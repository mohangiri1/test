def one_plus_erf(x):
    """
    Compute a numerically stable implementation of the function `1 + erf(x)` using the cumulative distribution function (CDF) of the standard normal distribution.

    Mathematically
        1 + erf(x) = 2 * CDF(x * sqrt(2))

    Parameters
    ----------
    x : float or array-like
          Input value(s) at which to evaluate the function.

    Returns
    ----------
    one_plus_erf : float or ndarray
          The computed values of the function `1 + erf(x)` for the input `x`.
    """
    return 2. * spst.norm.cdf(x, scale=1.0 / np.sqrt(2.0))

    @staticmethod
    def instrument_response(sigma, t):
        """
        Define an instrument response function (IRF) as a Gaussian.

        Returns
        ----------
        array-like
            Constant function values.
        """
        return np.exp(-t**2 / (2*sigma**2)) / (sigma * np.sqrt(2*np.pi))

        Parameters

    @staticmethod
    def TransDirect_frt_Func(x, t0, sig, T, T_A):
        """
        Transient function with finite rise time.

        Parameters
        ----------
            x : array-like
                Independent variable, pump probe time delay [ps] .
            t0 : float
                Time offset [ps].
            sig : float
                Overal temporal profile of the pump and probe pulse [ps].
            T : float
                Decay time [ps].
            T_A : float
                Finite rise time parameter [ps].

        Returns
        ----------
        array-like
            Transient function values.
        """
        if T > 0.02 * sig:
            t_eq = T * T_A / (T + T_A)
            trans = 0.5*np.exp(-(x - t0) / T) * one_plus_erf((+(x - t0) / (np.sqrt(2)*sig)) - (0.5 * (np.sqrt(2)*sig) / T)) * np.exp(0.25 * ((np.sqrt(2)*sig) / T) ** 2)
            rise_time_response = 0.5*np.exp(-(x - t0) / t_eq) * one_plus_erf((+(x - t0) / (np.sqrt(2)*sig)) - (0.5 * (np.sqrt(2)*sig) / t_eq)) * np.exp(0.25 * ((np.sqrt(2)*sig) / t_eq) ** 2)
            return trans - rise_time_response
        else:
            return Fitting_Basis.gauss_func(x, t0, sig)

class Models:
    """
    Collection of custom theoretical model functions for fitting data.

    This class provides various theoretical model functions for fitting experimental data. 
    These functions cover scenarios such as single exponential decay, double exponential decay, 
    triple exponential decay, and tetra exponential decay, with options for finite rise time in some cases.
    """

    @staticmethod
    def Select_model(model_type):
        """
        Select and return the appropriate model function based on the model type.

        Parameters
        ----------
            model_type : str
                Type of model.

        Returns
        ----------
            function
                The selected model function.
        """
        if model_type == 'single_exp':
            return Models.single_exp
        elif model_type == 'double_exp':
            return Models.double_exp
        elif model_type == 'triple_exp':
            return Models.triple_exp
        elif model_type == 'exp_switch_exp':
            return Models.exp_switch_exp
        elif model_type == 'exp_switch':
            return Models.exp_switch
        elif model_type == 'frt_exp_switch':
            return Models.frt_exp_switch
        elif model_type == 'double_exp_gauss':
            return Models.double_exp_gauss
        elif model_type == 'tetra_exp':
            return Models.tetra_exp
        elif model_type == 'exp_1_frt':
            return Models.exp_1_frt
        elif model_type == 'exp_2_frt':
            return Models.exp_2_frt
        elif model_type == 'frt':
            return Models.frt
        elif model_type == 'exp_1frt_2_exp':
            return Models.exp_1frt_2_exp
        elif model_type == 'exp_1frt_3_exp':
            return Models.exp_1frt_3_exp
        elif model_type == 'thermal_frt':
            return Models.thermal_frt
        else:
            raise ValueError("Invalid model type. Supported model types are: 'single_exp', 'double_exp',double_exp_gauss,  'triple_exp', 'tetra_exp', 'exp_1_frt', 'exp_2_frt', 'frt', 'thermal_frt'.")

    @staticmethod
    def thermal_frt(t, A1, tau1, A2, offset, sig, t0, T_A):
        """
        Custom theoretical model function with finite rise time.

        Parameters
        ----------

            t : array-like
                Time data [ps].
            A1 : float
                Amplitude of the first exponential decay.
            tau1 : float
                Lifetime of the first exponential decay [ps].
            A2 : float
                Amplitude of the second exponential decay.
            offset : float
                Offset parameter.
            sig : float
                Overal temporal profile of the pump and probe pulse [ps].
            t0 : float
                Time offset [ps].
            T_A : float
                Finite rise time parameter [ps].

        Returns
        ----------
            array-like
                Model prediction.
        """
        model = A1 * Fitting_Basis.TransDirect_frt_Func(t, t0, sig, tau1, T_A) + \
                A2 * Fitting_Basis.TransDirectFunc(t, t0, sig, tau1) + \
                offset * Fitting_Basis.ConstFunc(t)
        return model

    def stretched_exponential(self, t, A, T, beta):
        """
        Custom stretched/compressed function.

        Parameters
        ----------
            t : float or array-like
                Time data [ps].
            A : float
                Amplitude.
            T : float
                Time constant [ps].
            beta : float
                Stretching/compression parameter.

        Returns
        ----------
            array-like
                Model prediction.
        """
        return self.heaviside(t) * A * np.exp(-(np.abs(t / T)) ** beta)

    def gaussian(self, x, mu, sigma):
        """
        Calculate the value of a Gaussian (normal) distribution.

        Parameters
        ----------
            x : float or numpy.ndarray
                The value(s) at which to evaluate the Gaussian distribution.
            mu : float
                The mean (average) value of the distribution.
            sigma : float
                The standard deviation, a measure of the spread of the distribution.

        Returns
        ----------
            float or numpy.ndarray
                The value(s) of the Gaussian distribution at the given point(s) x.
        """
        coefficient = 1 / (sigma * np.sqrt(2 * np.pi))
        exponent = -((x - mu) ** 2) / (2 * sigma ** 2)
        return coefficient * np.exp(exponent)

    def convolution_stretchedExp(self, t, tau, t0, A, sigma, beta):
        """
        Perform convolution of a stretched exponential decay function and a Gaussian function with given parameters using direct summation method.

        Parameters
        ----------
            t : numpy.ndarray
                Array of time points [ps].
            tau : float
                Time constant of the stretched exponential decay function.
            t0 : float
                Time offset [ps].
            A : float
                Amplitude of the stretched exponential decay function.
            sigma : float
                Standard deviation of the Gaussian function [ps].
            beta : float
                Stretching parameter of the stretched exponential decay function.

        Returns
        ----------
            numpy.ndarray
                Result of convolution of the two functions.
        """
        # Alias t to x and t0 to mu for better readability
        x = t
        mu = t0
        
        # Direct numerical convolution of Gaussian with stretched exponential decay
        IRF = self.gaussian(x - x.mean(), mu, sigma)
        IRF /= IRF.sum()  # Area normalization
        
        decay = self.stretched_exponential(t, A, tau, beta)
        
        return np.convolve(decay, IRF, mode="same")
