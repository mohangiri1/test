class Fitting:
    """
    Class for performing data fitting using bootstrap optimization and differential evolution.

    Attributes
    ----------
    num_bootstraps : int
        Number of bootstrap iterations for optimization to estimate parameter uncertainties.
    selected_model_type : str
        The type of model used for fitting, selected from predefined model options.
    Bound_class : instance, class
        Helps in getting the bounds i.e max and min value of the parameters between which optimization is to be performed.
    loss : str
        Loss function to use ('mse', 'mae', 'chi_square', 'tukey', 'least_squares', or 'ridge').
    MinMaxScaling : bool, optional
        If `True`, applies Min-Max scaling to the `Thz` data before fitting. Default is `False`.
    track_progress : bool, optional
        If `True`, tracks and displays optimization progress. Default is `False`.
    save_file_path : str, optional
        Path to save intermediate results or progress files. Default is `'./'`.
    plot_within_bounds : bool, optional
        Default is False, if true, plots the track progress within the bounds.

    Notes
    -----
        - The function supports multiple pre-defined model types for fitting. 
        - Parameter bounds are dynamically defined based on input data (`Thz`) and are adjusted depending on the model type.
        - The fitting procedure utilizes differential evolution for global optimization within bootstrap iterations to minimize the objective function.

    Examples
    --------
    >>> # Fit a double exponential decay model to simulated pump-probe data:
    >>> import numpy as np
    >>> t = np.linspace(0, 100, 1000)
    >>> Thz = np.exp(-(t-10) / 30) + 0.5 * np.exp(-(t-10) / 10) + np.random.normal(0, 0.1, size=t.shape)
    >>> num_bootstraps = 100
    >>> selected_model_type = 'double_exp'
    >>> Bounds_class = Bounds()
    >>> fitting_func = Fitting(num_bootstraps, selected_model_type, Bounds_class, MinMaxScaling= True, track_progress = False, save_file_path = './')
    >>> fitted_params, uncertainties, final_model_mean = fitting_func.bootstrap_optimization(t, Thz)
    """

    def __init__(self, num_bootstraps, selected_model_type, Bounds_class, loss = 'mae'
                 , MinMaxScaling = False, track_progress = True, save_file_path = './', plot_within_bounds = False):
        '''
        Initialize the Fitting instance.
        '''
        self.num_bootstraps = num_bootstraps
        self.selected_model_type = selected_model_type
        self.theoretical_model =  Models.Select_model(self.selected_model_type)
        self.loss = loss
        self.track_progress = track_progress
        self.save_file_path = save_file_path
        self.MinMaxScaling = MinMaxScaling
        self.Bounds_class = Bounds_class
        self.param_names = Models_Utilities(self.selected_model_type).get_ParamName()
        self.plot_within_bounds = plot_within_bounds

    def objective_function(self, params, t, Thz):
        """
        Compute the objective function to minimize during optimization.

        Parameters
        ----------
        params : array-like
            Model parameters to optimize.
        t : array-like
            Time data.
        Thz : array-like
            Experimental THz data.

        Returns
        -------
            float
                Value of the loss function for the given parameters.
        """
        model = self.theoretical_model(t, *params)
        residuals = Thz - model
        if self.loss == 'mse':
            # Mean Squared Error (MSE)
            return np.mean((residuals)**2)
        
        elif self.loss == 'least_squares':
            # Least Squares
            return np.sum((residuals)**2)
        
        elif self.loss == 'chi_square':
            # Chi-square
            return np.sum(residuals**2 / Thz.var())
        
        elif self.loss == 'mae':
            # Mean Absolute Error (MAE)
            return np.mean(np.abs(residuals))
        
        elif self.loss == 'tukey':
            # Tukey's Biweight Loss
            return np.sum(self.tukey_biweight_loss(residuals, c=4))
        
        elif self.loss == 'least_squares':
            # Least Squares
            return np.sum((residuals)**2)
        
        elif self.loss == 'ridge':
            alpha=0.01
            # Ridge Regression
            return np.sum((residuals)**2) + alpha * np.sum(params**2)  # Ridge penalty term added
        else:
            raise ValueError("Invalid loss function. Choose 'mse', 'mae', 'chi_square', 'tukey', 'least_squares', or 'ridge'.")

    def tukey_biweight_loss(self, residuals, c):
        """
        Compute Tukey's biweight loss function for residuals.

        Parameters
        ----------
        residuals : array-like
            Differences between observed and predicted values.
        c : float
            Tuning constant for the loss calculation.

        Returns
        -------
        array-like
            Tukey's biweight loss values for residuals.
        """
        scaled_residuals = residuals / c
        mask = np.abs(scaled_residuals) <= 1
        return (1 - scaled_residuals**2)**2 * mask

    def bootstrap_optimization(self, t, Thz):
        """
        Fit the given time-dependent pump-probe data (``t``, ``Thz``) using a specified model type and bootstrapping optimization.

        Parameters
        ----------
        t : numpy.ndarray
            Array of time points.
        Thz : numpy.ndarray
            Array of measured data points corresponding to the time array `t`.

        Returns
        -------
        fitted_params : numpy.ndarray
            Optimized parameters for the selected model, averaged over bootstrap samples.
        uncertainties : numpy.ndarray
            Estimated uncertainties of the fitted parameters based on bootstrap results.
        final_model_mean : numpy.ndarray
            The mean of all bootstrap model predictions, representing the final fitted model.
        """
        if self.MinMaxScaling:
            Thz = Fitting.MinMaxscaler(Thz)

        self.bounds = self.Bounds_class.bound_var(t, Thz, self.selected_model_type)
        # Initialize arrays to store optimized parameters from each bootstrap
        optimized_params_list = []

        for _ in range(self.num_bootstraps):
            # Generate a bootstrap sample by randomly sampling with replacement
            indices = np.random.choice(len(t), len(t), replace=True)
            t_bootstrap, Thz_bootstrap = t[indices], Thz[indices]
            result = differential_evolution(self.objective_function
                                            , self.bounds
                                            , args=(t_bootstrap, Thz_bootstrap)
                                            , popsize=5
                                            , seed=50
                                            , strategy='best1bin'
                                            , disp=False
                                            , atol=1e-12
                                            , polish=True)
                                            
            # Store the optimized parameters
            optimized_params_list.append(result.x)

        # Convert list of results to a numpy array for easier manipulation
        self.optimized_params_array = np.array(optimized_params_list)

        # Calculate median and 95% confidence intervals (2.5th and 97.5th percentiles) for each parameter to select the best fit.
        self.fitted_params = np.median(self.optimized_params_array, axis=0)
        self.lower_bounds = np.percentile(self.optimized_params_array, 2.5, axis=0)
        self.upper_bounds = np.percentile(self.optimized_params_array, 97.5, axis=0)
        self.confidence_intervals = np.vstack((self.lower_bounds, self.upper_bounds)).T

        # Calculate uncertainties as half the width of the 95% confidence intervals
        self.uncertainties = (self.upper_bounds - self.lower_bounds) / 2

        # Use the  fitted_params to generate the fitted_value
        fitted_value = self.theoretical_model(t, *self.fitted_params)
        self.fitted_value = np.array(fitted_value)            

        # Display the optimized parameters with confidence intervals
        self.display_result()

        # Print the metric Score of fitting:
        metric = Goodness_of_fit(Thz, self.fitted_value, self.fitted_params, self.uncertainties)
        metric.display_goodness_of_fit()

        self.plot_fit(t, Thz, self.fitted_value)
        self.plot_components(t)

        # Plot the optimized parameters if needed controlled by Flag self.track_progress
        if self.track_progress:
            #self.plot_optimized_parameters()
            self.box_plot_optimized_parameters()

        return self.fitted_params, self.uncertainties, self.fitted_value

    def VariableDependentFit(self, df_experimental, unit, variable_name
                            , variable_value, energy = None, overrides = None):
        """
        Perform variable dependent fitting (eg power dependent or temperature dependent study) for a given variable (e.g., power, temperature, energy).

        Parameters
        ----------
        df_experimental : pandas.DataFrame
            The input experimental data with specific column requirements:
            - Must contain a `time_delay [ps]` column representing the independent variable (time delay).
            - Must contain columns named with the format `del_T (<value><unit>)`, where `<value>` corresponds to the variable 
            (e.g., power or temperature) and `<unit>` is the unit (e.g., `mW`, `K`).
            - Optionally include an `Energy [eV]` column for energy-specific data.
        
        unit : str
            The unit of the variable. Options include:
            - `'mW'` for power-dependent fitting
            - `'K'` for temperature-dependent fitting
            - `'eV'` for energy
            - `'nm'` for wavelength
        
        variable_name : str
            Name of the variable for fitting (e.g., `"power"`, `"Temperature"`, `"Energy"`, `"Wavelength"`).

        variable_value : list or numpy array
            A list or array of variable values for which the fitting is to be performed. These values must match 
            the numeric values in the `del_T (<value><unit>)` column names of the input `df_experimental`.

        energy : float or int, optional
            An optional energy value to include in the resulting parameter DataFrame.

        overrides : list of dictionary
            An optional dictionary to modify the parameter space for a given model.

        Returns
        --------
        df_with_result : pandas.DataFrame
            A DataFrame containing:
            - Original experimental data (`time_delay [ps]` and `del_T (<value><unit>)` columns).
            - Fitted values for each variable as new columns named `del_T (<value><unit>)_fit`.
            - `Energy [eV]` (if available) and the selected model type as additional columns.

        df_param : pandas.DataFrame
            A DataFrame containing extracted parameters and their uncertainties for each variable value. 
            Includes:
            - The variable name and its values.
            - Fitted parameters and their uncertainties.
            - Energy (if specified) and the selected model type.

        Notes
        -----
            - Iterates through the ``variable_value`` list and performs fitting for each value.
            - The fitting process uses ``bootstrap_optimization`` to extract fitted parameters, uncertainties, 
            and fitted values.
            - The ``SaveResults`` class is used to generate parameter DataFrames.
        """

        #Create empty dataframe for all the fitted data: Also Create empty list of fitted params for all the data
        df_with_result = pd.DataFrame([])
        fitted_params_list = []
        uncertainties_list = []

        first_name = self.save_file_path
        for i, value in enumerate(variable_value):
            self.save_file_path = f'{first_name}_{str(int(value))}{unit}'

            t = df_experimental[f'time_delay [ps]({value}{unit})']
            Thz = df_experimental[f'del_T ({value}{unit})']

            if overrides:
                self.Bounds_class = BoundsClass(overrides[i])
                
            print(f'Fitting Data for {variable_name} = {value} {unit}')
            fitted_params, uncertainties, fitted_value = self.bootstrap_optimization(t, Thz)
            fitted_params_list.append(fitted_params)
            uncertainties_list.append(uncertainties)
            
            # Create a temporary DataFrame for this variable_value
            temp_df = pd.DataFrame({
                f'time_delay [ps]({value}{unit})': t,
                f'del_T ({int(value)}{unit})': Thz,
                f'del_T ({int(value)}{unit})_fit': fitted_value
            })

            # Merge with the main DataFrame using an outer join to handle mismatched rows
            if df_with_result.empty:
                df_with_result = temp_df
            else:
                df_with_result = pd.merge(df_with_result, temp_df, how='outer', left_index=True, right_index=True)

        # Add constant columns to the result DataFrame
        df_with_result['Model'] = self.selected_model_type
        if 'Energy [eV]' in df_experimental.columns:
            df_with_result['Energy [eV]'] = df_experimental['Energy [eV]'].iloc[0]  # Assuming constant energy for all rows

        # Generate the parameters DataFrame
        saveresult = SaveResults(self.selected_model_type)
        df_param = saveresult.dataFrameofResults_withVariable(variable_name, variable_value, fitted_params_list, uncertainties_list)

        # Add metadata to parameters DataFrame
        if energy is not None:
            df_param['Energy [eV]'] = energy
        df_param['Model'] = self.selected_model_type

        return df_with_result, df_param

    def plot_fit(self, t, Thz, final_model_mean):
        '''
        Visualize the fitted model and residuals compared to the experimental data.

        Parameters
        ----------
        t : array-like
            Time data.
        Thz : array-like
            Experimental THz data.
        final_model_mean : array-like
            Model values computed with the optimized parameters.
        '''
        fig, ax = plt.subplots(1, 2, figsize = (10,6))
        ax[0].plot(t, Thz, 'o', label = 'Data')
        ax[0].plot(t, final_model_mean, '-', label = 'Fit')
        ax[0].set_xlabel('t [ps]')
        ax[0].set_ylabel(r'$\Delta T $ [a.u.]')
        ax[0].legend()

        ax[1].plot(t, Thz - final_model_mean, 'o', label = 'Residual')
        ax[1].set_xlabel('t [ps]')
        ax[1].set_ylabel(r'$\Delta T$ [a.u.]')
        plt.legend()
        plt.tight_layout()
        if self.track_progress:
            plt.savefig(f'{self.save_file_path}_fited_result.pdf')
        plt.show()

    def display_result(self):
        """
        Display the optimized parameters along with their uncertainties and confidence intervals.
        """
        # Get parameter names for the selected model type
        header = self.param_names

        # Ensure the number of parameter names matches the fitted parameters
        if len(header) != len(self.fitted_params):
            raise ValueError("Mismatch between number of parameter names and fitted parameters.")

        # Prepare the table data: parameter names, estimated values, and uncertainties
        table_data = np.column_stack((header, self.fitted_params, self.uncertainties))

        #Print the results:
        print(f'Results from the Fitting using {self.selected_model_type} Model:')
        print('-' * 70)
        # Print the table header
        print('{:<15} {:<25} {:<25}'.format('Parameter', 'Estimated Value', 'Uncertainty'))
        print('-' * 70)

        # Display the table rows
        for row in table_data:
            param_name = row[0]
            mean_val = row[1]  # Estimated value
            std_val = row[2]   # Uncertainty
            std_val_numeric = float(std_val)  # Convert uncertainty to a numeric type
            std_sign = '' if std_val_numeric >= 0 else ''
            std_val_str = f'{std_sign} {abs(std_val_numeric)}'
            print('{:<15} {:<25} {:<25}'.format(param_name, str(mean_val), std_val_str))

    def plot_optimized_parameters(self):
        """
        Visualize optimized parameters across bootstrap iterations and pairwise relationships between parameters.
        """
        # Get parameter names for the selected model type
        param_names = self.param_names
        num_params = len(param_names)

        # Calculate the total number of subplots needed
        total_subplots = num_params + num_params * (num_params - 1) // 2

        # Calculate the number of rows and columns needed for plotting
        num_rows = (total_subplots + 1) // 2  # Add 1 to handle odd number of subplots
        num_cols = 2

        # Create subplots
        fig, axes = plt.subplots(num_rows, num_cols, figsize=(12, 4 * num_rows))
        axes = axes.flatten()

        # Plot each parameter vs. the number of bootstraps
        for i in range(num_params):
            ax = axes[i]
            param_values = self.optimized_params_array[:, i]
            ax.plot(range(1, self.num_bootstraps + 1), param_values, '-o', label=f'{param_names[i]} Samples')

            # Plot median and confidence interval
            ax.axhline(y=self.fitted_params[i], color='r', linestyle='--', label=f'Median {param_names[i]}')
            ax.axhline(y=self.lower_bounds[i], color='b', linestyle='--', label=f'2.5th Percentile (CI) {param_names[i]}')
            ax.axhline(y=self.upper_bounds[i], color='b', linestyle='--', label=f'97.5th Percentile (CI) {param_names[i]}')

            # Set y-limits based on bounds if available
            if self.bounds is not None and self.plot_within_bounds == True:
                ax.set_ylim(self.bounds[i])

            ax.set_ylabel(param_names[i])
            ax.legend()

        # Plot each parameter against another parameter
        start_param = num_params
        for i in range(num_params):
            for j in range(i + 1, num_params):
                if start_param < total_subplots:
                    ax = axes[start_param]
                    param_values_i = self.optimized_params_array[:, i]
                    param_values_j = self.optimized_params_array[:, j]
                    ax.plot(param_values_i, param_values_j, 'o')

                    # Set x- and y-limits based on bounds if available
                    if self.bounds is not None and self.plot_within_bounds == True:
                        ax.set_xlim(self.bounds[i])
                        ax.set_ylim(self.bounds[j])

                    ax.set_xlabel(param_names[i])
                    ax.set_ylabel(param_names[j])
                    start_param += 1

        # Hide any remaining empty subplots
        for i in range(start_param, total_subplots):
            fig.delaxes(axes[i])

        plt.tight_layout()
        plt.savefig(f'{self.save_file_path}_fit_statistics.pdf')
        plt.show()

    @staticmethod
    def find_t0_max_min(t, Thz):
        """
        Identify the time range near the minimum signal value where the THz signal reaches within a threshold.

        Parameters
        ----------
        t : array-like
            Time data.
        Thz : array-like
            Experimental THz data.
            
        Returns
        -------
        tuple
            A tuple of ``t0_max`` (time at the minimum signal) and ``t0_min`` (time near the threshold).
        """
        min_index = np.argmin(Thz)  # Find the index of the minimum value in Thz
        t0_max = t[min_index]
        threshold = min(Thz) * 0.03  # Define the threshold (3% below the mean)

        # Iterate backward from the minimum index to find where Thz first reaches within 3% of the mean
        for i in range(min_index, -1, -1):
            if Thz[i] >= threshold:
                t0_min = t[i]
                return t0_max, t[i]  # t0_min and t0_max

        return t0_max, 20  # Return 20 if no such time is found

    @staticmethod
    def MinMaxscaler(data):
        """
        Scale data to the range ``[-1, 0]``.

        Parameters
        ----------
        data : array-like
            Input data to be scaled.

        Returns
        -------
        array-like
            Scaled data in the range [-1, 0].
        """
        return ( data - min(data) ) /  ( max(data) - min(data) ) - 1

    def plot_components(self, t):
        """

        if self.selected_model_type == 'single_exp':

            comp_1 = A1 * Fitting_Basis.TransDirectFunc(t, t0, sig, tau1)
