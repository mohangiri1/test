    def get_rawdata(self, filename
                    , Normalize: bool = False
                    , from_zero: bool = False
                    , baseline_subtract: bool = False
                    , limit_time_delay: bool = False
                    , max_time = 100):
        """
        Read raw data from a CSV file and return Time and Field arrays.

        This function reads raw data from a CSV file and extracts the Time and Field arrays. Optionally, it allows normalization
        of the Lockin_X data, shifting the Time array to start from zero based on the minimum value of the Field array, and baseline subtraction.

        Parameters
        ----------
            filename : str
                Path to the CSV file containing raw data.
            Normalize : bool, optional
                If True, normalizes the Lockin_X data. Defaults to False.
            from_zero : bool, optional
                If True, shifts the Time array to start from zero based on the minimum value of the Field array. Defaults to False.
            baseline_subtract : bool, optional
                If True, performs baseline subtraction on the Lockin_X data. Defaults to False.
            limit_time_delay : bool, optional
                If True, limits maximum time delay and on the Lockin_X data. Defaults to False.
            max_time : float, optional
                Provide the maximum time delay while doing the analysis.

        Returns
        -------
            tuple
                A tuple containing Time and Field arrays.
        """

        raw_data = pd.read_csv(filename)  # Read raw text files
        Time = np.array(raw_data['Time (ps)'])
        # Assuming raw_data is a dictionary containing 'Lockin_X' data
        lockin_x = raw_data['Lockin_X']

        if baseline_subtract:
            # Calculate the baseline by averaging the specified range of data points
            baseline = np.mean(lockin_x[1:13])

            # Remove the baseline
            lockin_x_corrected = lockin_x - baseline
        else:
            lockin_x_corrected = lockin_x

        if Normalize:
            Field = np.array((lockin_x_corrected - min(lockin_x_corrected))/np.abs(max(lockin_x_corrected) - min(lockin_x_corrected)) - 1)
        else:
            Field = np.array(lockin_x_corrected)
            
        if from_zero:
            mask = (Time >= Time[np.argmin(Field)])
            Time = Time[mask] - Time[np.argmin(Field)]
            Field = Field[mask]
            
        if limit_time_delay:
            mask = (Time <= max_time)
            Time = Time[mask]
            Field = Field[mask]
        
        return Time, Field

    def select_column_for_a_value(self, variable_value, columns, unit='mW'):
        """
        Extract the column name from a list of columns where the specified variable value matches the numeric value associated with a unit (e.g., '15mW') in the column name.

        Parameters
        ----------
        variable_value : float or int
            The numeric value to search for in the column names (e.g., 15 if looking for '15mW').
        columns : list of str
            The list of column names to search through.
        unit : str, optional
            Default='mW', The unit that the numeric value is associated with in the column names.

        Returns
        -------
        str or None
            The matching column name if found, or None if no match is found.
        """
        for col in columns:
            if unit in col:
                try:
                    # Extract numeric value before the unit and convert it to float
                    value_in_column = float(col.split("(")[-1].split(unit)[0])
                    if value_in_column == float(variable_value):
                        return col
                except (ValueError, IndexError):
                    # Handle cases where parsing fails or column name format is unexpected
                    print(f"Could not parse column: {col}")
        print(f"No column found for value {variable_value} with unit {unit}.")
        return None

    def extract_value_from_column_name(self, columns, unit = 'mW', variable_name = 'Power'):
        """
        Extract power values in mW from column names and returns them as a pd.Series of floats.
        
        Parameters
        ----------
            columns : pd.Index or list
                Column names of a DataFrame.
            
        Returns
        -------
            pd.Series
                A Series containing power values as floats.
        """
        # List to store extracted power values
        variable_values = []

        for col in columns:
            # Check if "mW" is in the column name
            if unit in col:
                # Extract the numeric value before "mW" and convert to float
                try:
                    value = float(col.split("(")[-1].split(unit)[0])
                    variable_values.append(value)
                except ValueError:
                    # Handle cases where parsing fails
                    print(f"Could not extract power from column: {col}")
        
        # Return as a Pandas Series
        return pd.Series(variable_values, name=f"{variable_name} ({unit})")

    def data_file_csv_generator(self, filenames, variable_values, variable_unit='K', wavelength=None, energy=None, power=None):
        '''
        Generate a pandas DataFrame of `t` (time delay) and `Thz` values for various powers at a given wavelength, even when the `time_delay` values and rows in the files don't match.

        Parameters
        ----------
            filenames : list
                List of file paths to the data files.
            variable_values : list
                List of variable values corresponding to each file.
            variable_unit : str
                Unit of the variable values (default: 'K').
            wavelength : float, optional
                Wavelength (nm) to include in the DataFrame.
            energy : float, optional
                Energy (eV) to include in the DataFrame.
            power : float, optional
                Power (mW) to include in the DataFrame.

        Returns
        -------
            pd.DataFrame 
                DataFrame With combined `time_delay` and aligned `Thz` values from all files.
        '''

        combined_df = pd.DataFrame()  # Initialize an empty DataFrame

        for i, file in enumerate(filenames):
            # Load the time and Thz data for the current file
            t, Thz = self.get_rawdata(file, Normalize=False)

            # Create a DataFrame for this file
            temp_df = pd.DataFrame({
                f'time_delay [ps]({variable_values[i]}{variable_unit})': t,
                f'del_T ({variable_values[i]}{variable_unit})': Thz
            })

            # Merge with the combined DataFrame using an outer join
            if combined_df.empty:
                combined_df = temp_df
            else:
                combined_df = pd.merge(combined_df, temp_df, how='outer', left_index=True, right_index=True)

        # Add optional metadata columns
        if wavelength is not None:
            combined_df['Wavelength [nm]'] = wavelength
        if energy is not None:
            combined_df['Energy [eV]'] = energy
        if power is not None:
            combined_df['Power [mW]'] = power

        # Sort by time_delay for better readability
        combined_df = combined_df.sort_index().reset_index(drop=True)

        return combined_df
