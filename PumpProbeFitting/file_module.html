    @staticmethod
    def ExtractFileNames(energy_value, filename_df, user_prefix, unit = 'K', variable_name = 'Temperature', extra_parameter = 'Power', extra_param_unit = 'mW'):
        """
        Extract relevant information from a DataFrame based on a given energy value.

        Parameters
        -----------
        energy_value : float
            The specific energy value to filter the DataFrame.
        filename_df : pandas.DataFrame
            The DataFrame containing information about filenames and associated data.
        user_prefix : str
            A string to be prepended to the extracted filenames.
        unit : str
            Unit for the `variable_name` (defaults to 'K').
        variable_name : str
            The name of the variable to extract (defaults to 'Temperature').
        extra_parameter : str
            The name of an extra parameter to extract (defaults to 'Power').
        extra_param_unit : str
            Unit for the `extra_parameter` (defaults to 'mW').

        Returns
        --------
            `variable_values`: list
                A list of values for the specified `variable_name`.
            `filenames`: list
                A list of full file paths constructed using `user_prefix` and the extracted filenames.
            `extra_parameter_value`: list
                A list of unique values for the `extra_parameter`.
        """

        # Filter rows where 'energy [eV]' matches energy_value
        filtered_df = filename_df[filename_df['energy [eV]'] == energy_value]

        # Extract variable_values and reset the index
        variable_values = filtered_df[f'{variable_name} [{unit}]'].reset_index(drop=True).tolist()

        # Extract filename suffixes and build full file paths
        filename_suffix = filtered_df['filename']
        filenames = [f'{user_prefix}{suffix}' for suffix in filename_suffix]

        # Extract unique power values
        extra_parameter_value = filtered_df[f'{extra_parameter} [{extra_param_unit}]'].unique().tolist()

        return variable_values, filenames, extra_parameter_value

    @staticmethod
    def generate_path():
        """
        Create the necessary directories for storing DataFrames and figures.

        Returns
        --------
            `DataFrame_filepath`: pandas.DataFrame
                The path to the directory where DataFrames will be saved.
            `figures_filepath`: str
                The path to the directory where figures will be saved.
        """
        # DataFrame directory where the filenameinfo.csv is located. Also, it is the df saving directory.
        DataFrame_filepath = './../dataframes/'
        # Check if the directory exists, if not, create it
        if not os.path.exists(DataFrame_filepath):
            os.makedirs(DataFrame_filepath)

        # Figure Directory:
        figures_filepath = './../figures/'
        # Check if the directory exists, if not, create it
        if not os.path.exists(figures_filepath):
            os.makedirs(figures_filepath)
        return DataFrame_filepath, figures_filepath

    def generate_progress_path(figures_filepath, energy):
        """
        Create a directory for storing intermediate figures during the fitting process.

        Parameters
        -----------
            figures_filepath : str
                The base path for figure directories.
            energy : float
                The energy value associated with the fitting process.

        Returns
        --------
            progress_location : str
                The path to the directory for storing intermediate fitting figures.
        """
        # Location where intermediate figures of fitting are to be stored. Depends upon figures_filepath        
        progress_location = f'{figures_filepath}fitting_statistics/{energy}eV/'
        # Check if the directory exists, if not, create it
        if not os.path.exists(progress_location):
            os.makedirs(progress_location)
        return f'{progress_location}{energy}eV'
