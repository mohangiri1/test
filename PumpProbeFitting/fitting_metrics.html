class Goodness_of_fit:
    """
    A class to evaluate and display goodness-of-fit metrics for a fitted model.

    This class calculates the following goodness-of-fit metrics:
    - Chi-Square (χ²)
    - Reduced Chi-Square
    - Akaike Information Criterion (AIC)
    - Bayesian Information Criterion (BIC)
    - Coefficient of Determination (R²)

    Attributes
    ----------
    actual_data : numpy.ndarray
        Observed data points used for fitting.
    fitted_data : numpy.ndarray
        Predicted data points from the fitted model.
    fitted_params : numpy.ndarray
        Parameters of the fitted model.
    uncertainties : numpy.ndarray
        Uncertainties associated with each fitted parameter.

    Notes
    -----
    - Chi-Square (χ²) and Reduced Chi-Square are absolute measures of goodness of fit.
    - AIC and BIC allow comparison between models, with lower values indicating better fit 
      after penalizing for model complexity.
    - R² is a relative measure often used for quick assessment of fit quality.

    References
    ----------
    Based on lmfit documentation: https://lmfit.github.io/lmfit-py/fitting.html

    Examples
    --------
    >>> import numpy as np
    >>> actual_data = np.array([1.0, 2.0, 3.0, 4.0])
    >>> fitted_data = np.array([1.1, 1.9, 3.2, 3.8])
    >>> fitted_params = np.array([0.5, 0.2])
    >>> uncertainties = np.array([0.1, 0.05])
    >>> goodness = Goodness_of_fit(actual_data, fitted_data, fitted_params, uncertainties)
    >>> goodness.display_goodness_of_fit()
    --------------------------------------------------------------------------------
    Goodness of Fit Metrics
    --------------------------------------------------------------------------------
    Metric                                     Value
    --------------------------------------------------------------------------------
    Chi-Square (χ²)                               0.0476
    Reduced Chi-Square                             0.0159
    Akaike Information Criterion (AIC)            -0.2534
    Bayesian Information Criterion (BIC)          -0.1004
    Coefficient of Determination (R²)              0.9987
    --------------------------------------------------------------------------------
    """
    def __init__(self, actual_data, fitted_data, fitted_params, uncertainties = None):
        self.actual_data = actual_data
        self.fitted_data = fitted_data
        self.fitted_params = fitted_params
        self.uncertainties = uncertainties

    def chi_square(self):
        """
        Calculate the chi-square value.

        Returns
        -------
        chi_square : float
            Chi_square value.
        """
        residual_square = (self.actual_data - self.fitted_data)**2
        return np.sum(residual_square)

    def reduced_chi_square(self):
        """
        Calculate the reduced chi-square value.

        Returns
        -------
        reduced_chi_square : float
            Reduced chi_square value.
        """
        chi_sq = self.chi_square()
        N = self.actual_data.size
        N_params = len(self.fitted_params)
        return chi_sq / (N - N_params)

    def AkaikeInformationCriterion(self):
        """
        Calculate the Akaike Information Criterion (AIC).

        Returns
        -------
        aic : float
            Akaike Information Criterion (AIC).
        """
        chi_sq = self.chi_square()
        N = self.actual_data.size
        N_params = len(self.fitted_params)
        aic = N * np.log(max(chi_sq / N, 1e-10)) + 2 * N_params
        return aic

    def BayesianInformationCriterion(self):
        """
        Calculate the Bayesian Information Criterion (BIC).

        Returns
        -------
        bic : float
            Bayesian Information Criterion (BIC).
        """
        chi_sq = self.chi_square()
        N = self.actual_data.size
        N_params = len(self.fitted_params)
        bic = N * np.log(max(chi_sq / N, 1e-10)) + np.log(N) * N_params
        return bic

    def r_squared(self):
        """
        Calculate the coefficient of determination (R²).

        Returns
        -------
        r_square : float
            R_square value.
        """
        ss_res = np.sum((self.actual_data - self.fitted_data)**2)  # Residual sum of squares
        ss_tot = np.sum((self.actual_data - np.mean(self.actual_data))**2)  # Total sum of squares
        return 1 - (ss_res / ss_tot)

    def display_goodness_of_fit(self):
        """
        Display goodness-of-fit metrics in a well-formatted table.
        """
        chi_sq = self.chi_square()
        reduced_chi_sq = self.reduced_chi_square()
        aic = self.AkaikeInformationCriterion()
        bic = self.BayesianInformationCriterion()
        r2 = self.r_squared()

        # Display table
        print('\n' + '=' * 80)
        print('Goodness of Fit Metrics')
        print('-' * 80)
        print(f'{"Metric":<40} {"Value":>25}')
        print('-' * 80)
        print(f'{"Chi-Square (χ²)":<40} {chi_sq:>25.4f}')
        print(f'{"Reduced Chi-Square":<40} {reduced_chi_sq:>25.4f}')
        print(f'{"Akaike Information Criterion (AIC)":<40} {aic:>25.4f}')
        print(f'{"Bayesian Information Criterion (BIC)":<40} {bic:>25.4f}')
        print(f'{"Coefficient of Determination (R²)":<40} {r2:>25.4f}')
