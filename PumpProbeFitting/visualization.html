class PowerDependentResults:
    '''
    Class to visualize and analyze the results obtained from power-dependent pump-probe experiments at a fixed pumped energy.

    Attributes
    ----------
    save_figure : bool
        If True, saves the generated figures.
    file_path : str
        The file path and prefix for saving figures.
    scale_fluence : bool
        If True, scales power (in mW) to fluence (in µJ/cm²).
    beam_diameter : float
        Beam diameter of the pump laser in centimeters (cm).
    rep_rate : float
        Repetition rate of the pump laser in Hertz (Hz).

    Notes
    ------
    - Power should be in milliwatts (mW).
    - Lifetimes (taus) should be in picoseconds (ps).
    - Amplitudes (A) can be in arbitrary units (a.u.).
    - Repetition rate (rep_rate) should be in Hertz (Hz).
    - Beam diameter should be provided in centimeters (cm).
    '''
    def __init__(self, save_figure = True, file_path = './', scale_fluence = False
                 , beam_diameter = 0.5, rep_rate = 1000):
        '''
        Initialize the PowerDependentResults class with user-defined attributes.

        Parameters
        ----------
        save_figure : bool, optional
            If True, saves all generated figures (default is True).
        file_path : str, optional
            File path and prefix for saving figures (default is './').
        scale_fluence : bool, optional
            If True, scales power to fluence (default is False).
        beam_diameter : float, optional
            Diameter of the pump beam in centimeters (default is 0.5).
        rep_rate : float, optional
            Repetition rate of the pump laser in Hertz (default is 1000).
        '''
        self.save_figure = save_figure
        self.file_path = file_path
        self.scale_fluence = scale_fluence
        self.beam_diameter = beam_diameter
        self.rep_rate = rep_rate

    def PowerVsParameter(self, parameter
                        , uncertainty
                        , power
                        , param_name: str = 'tau1'
                        , scale_amplitude = False
                        , symbol: str = '<'
                        , color:str = 'black'
                        , error_bar: bool = True):
        '''
        Plot power vs a specified parameter with options for error bars, scaling, and customization.

        Parameters
        ----------
        parameter : list or numpy.ndarray
            The parameter values (e.g., amplitude or lifetime) to be plotted.
        uncertainty : list or numpy.ndarray
            Uncertainty values associated with the parameter.
        power : list or numpy.ndarray
            Pump power values in milliwatts (mW).
        param_name : str
            The name of the parameter being plotted (e.g., 'tau1', 'A1').
        scale_amplitude : bool, optional
            If True, scales amplitudes for better visualization (default is False).
        symbol : str, optional
            Marker style for the plot (default is '<').
        color : str, optional
            Color for the plot (default is 'black').
        error_bar : bool, optional
            If True, includes error bars on the plot (default is True).

        Raises
        ------
        ValueError:
            If the param_name is not recognized.
        '''
        
        # check if parameter, uncertainty and power are list or numpy arrays.
        if type(parameter) == list: parameter = np.array(parameter)
        if type(uncertainty) == list: uncertainty = np.array(uncertainty)
        if type(power) == list: power = np.array(power)
        
        # Create a new figure
        plt.figure()
        
        # Scale the power if fluence scale is enabled
        if self.scale_fluence:
            x_unit = '$\mu J/cm^2$'
            # Calculate fluence for each power value
            fluences = self.calculate_fluence(power)
            # Round fluence values to 2 decimal places
            fluences = np.round(fluences, 2)
            xlabel = rf'Fluence [{x_unit}]'
        else:
            x_unit = 'mW'
            xlabel = f'Power [{x_unit}]'
            fluences = power
            
        # Define y-labels for parameters
        ylabel_map = {
            'A1': ('$A_1$', 'a.u.'),
            'A2': ('$A_2$', 'a.u.'),
            'A3': ('$A_3$', 'a.u.'),
            'A4': ('$A_4$', 'a.u.'),
            'tau1': ('$\\tau_1$', 'ps'),
            'tau2': ('$\\tau_2$', 'ps'),
            'tau3': ('$\\tau_3$', 'ps'),
            'tau4': ('$\\tau_4$', 'ps'),
            'T_A': ('$\\tau_A$', 'ps')
        }

        # Set the y-label based on param_name
        if param_name in ylabel_map:
            ylabel, unit = ylabel_map[param_name]
            ylabel = f'{ylabel} [{unit}]'
        else:
            raise ValueError(f"Unknown parameter name: {param_name}")
            
        if scale_amplitude:
            ylabel = f'$10^5 \\times ${ylabel}'
            if error_bar:
                # Plot A values with error bars
                plt.errorbar(fluences, 1e5*parameter, yerr=1e5*uncertainty, color=color, fmt=symbol)
            else:
                plt.plot(fluences, 1e5*parameter, symbol, color=color)
        else:
            if error_bar:
                # Plot A values with error bars
                plt.errorbar(fluences, parameter, yerr=uncertainty, color=color, fmt=symbol)
            else:
                plt.plot(fluences, parameter, symbol, color=color)
                
        plt.xlabel(xlabel)
        plt.ylabel(ylabel)
        
        # Format the y-axis to show 2 decimal places
        formatter = matplotlib.ticker.FuncFormatter(lambda x, pos: f'{x:.2f}')
        plt.gca().yaxis.set_major_formatter(formatter)
        plt.tight_layout()
        
        # Save the figure if specified
        if self.save_figure:
            plt.savefig(f'{self.file_path}_{param_name}.pdf')

    def calculate_fluence(self, power):
        '''
        Calculate the fluence of the pump laser.

        Parameters
        ----------
        power : list or numpy.ndarray
            Power values in milliwatts (mW).

        Returns
        -------
            numpy.ndarray
                Fluence values in µJ/cm².
        '''
        energy_per_pulse = power/self.rep_rate
        area = np.pi*self.beam_diameter**2/4
        fluence = energy_per_pulse*1e3/area
        return fluence

    @staticmethod
    def LogLog_LinearFit(x, y):
        '''
        Perform a log-log linear fit on the input data.

        Parameters
        ----------
        x : list or numpy.ndarray
            Independent variable values.
        y : list or numpy.ndarray
            Dependent variable values.

        Returns
        -------
        slope : float
            Fitted slope of the log-linear model.
        slope_uncertainty : float
            Uncertainty in the fitted slope.
        intercept : float
            Fitted intercept of the log-linear model.
        intercept_uncertainty : float
            Uncertainty in the fitted intercept.

        Notes
        ------
            - Absolute values are used for y if any of its elements are negative.
            - Prints a detailed fit report.
        '''
        x = np.array(x)
        y = np.array(y)

        # Define the model function
        def linear_model(x, slope, intercept):
            return slope * x + intercept
        
        # Check if any value in the parameters are -ve.
        is_negative = any(y_val<0 for y_val in y)
        if is_negative:
            print('Caution !!!!!!!!!!!!!!!!!!!!!!!!!!!!!')
            print('-' * 80)
            print(f'At least one value is negative in: \n{y}. \n log of negative value is not defined. \nPlease Consider checking the values. \n For Now performing the fitting using absolute values of each members from the list/array.')
            print('-' * 80)
        # Perform fitting using lmfit
        model = Model(linear_model)
        params = model.make_params(slope=1.5, intercept=-15)  # Initial guess for parameters
        result = model.fit(np.log(abs(y)), x=np.log(x), params=params)

        # Extract fitted parameters and uncertainties
        slope = result.params['slope'].value
        slope_err = result.params['slope'].stderr
        intercept = result.params['intercept'].value
        intercept_err = result.params['intercept'].stderr
        data = {'slope': slope, 'slope_uncertainty': slope_err, 'intercept': intercept, 'intercept_uncertainty': intercept_err}
        df = pd.DataFrame(data, index = [0])
        print('Fitted Results')
        print('-' * 80)
        print(result.fit_report())
        return slope, slope_err, intercept, intercept_err

    def Parameter_LogLinearFit_OriginalScale(parameter
                                               , variable_values
                                               , variable_name = 'Power'
                                               , variable_unit = 'mW'
                                               , param_name = 'tau1'
                                               , symbol: str = ['s', '--']
                                               , color:str = ['blue', 'green'], figure_location = None):
        '''
        Plot parameter vs variable_values with an overlay of log-linear fit on the original scale.

        plt.tight_layout()
        
        # Save the figure if specified
        if figure_location is not None:
            plt.savefig(f'{figure_location}_{param_name}_LogLinearFit_OriginalScale.pdf')

    @staticmethod
    def PlotExperimentalData_and_Fit(df_result, df_param, unit = 'mW', variable_name = 'Power', scale_amplitude = False, MinMaxScaling = False
                                     , save_file_name = None, colors = None, time_scale = True, include_fits = True):
        """
        Plot experimental data and corresponding fit for a dependent variable (e.g., power, temperature).

        Parameters
        ----------
        df_result : pandas.DataFrame
            DataFrame containing experimental data and their fits. 
            Required columns:
            - ``time_delay [ps]`` for time data.
            - Columns with names formatted as ``del_T (value unit)`` for experimental data.
            - Corresponding fit columns formatted as ``del_T (value unit)_fit``.
        unit : str
            Default ``'mW'``, Unit of the variable (e.g., ``'mW'`` for power, ``'K'`` for temperature).
        variable_name : str
            Default ``'Power'``, Name of the dependent variable being analyzed (e.g., ``'Power'``, ``'Temperature'``).
        scale_amplitude : bool
            Default False, If True, scales data and fit by a factor of \(10^5\) for better visualization.
        MinMaxScaling : bool
            Default False, If True, normalizes data and fit using Min-Max scaling.
        include_fits : bool
            Default True, if True, plots fit and experimental data both.

        Notes
        -----
        - Colors for data and their corresponding fits are matched for better clarity.
        - A colormap (``jet``) is used to differentiate between variable values.
        - If ``MinMaxScaling`` is enabled, data and fit values are scaled to ``[0, -1]``.

        Examples
        --------
        >>> # Example 1: Plot data and fit with default scaling
        >>> PlotExperimentalData_and_Fit(df_result)
        >>> # Example 2: Scale data and fit by a factor of 10^5
        >>> PlotExperimentalData_and_Fit(df_result, scale_amplitude=True)
        >>> # Example 3: Normalize data and fit to [0, -1]
        >>> PlotExperimentalData_and_Fit(df_result, MinMaxScaling=True)
        """
        data = Data()
        variable_value = data.extract_value_from_column_name(columns=df_result.columns, unit = unit, variable_name=variable_name).unique()
        variable_value = [int(value) for value in variable_value]
        t0_values = df_param['t0'].values

        if colors is None:
        # Generate a colormap based on the number of values
            colors = cm.jet(np.linspace(0, 1, len(variable_value)))

        for idx, value in enumerate(variable_value):
            plt.figure()
            col_Name_expt_data = f'del_T ({value}{unit})'
            col_Name_fit = f'{col_Name_expt_data}_fit'
            t0 = t0_values[idx] if time_scale else 0
            t = df_result[f'time_delay [ps]({value}{unit})'] - t0
            plt.xlim( min(t) + 2, max(t) )
            if scale_amplitude:
                plt.plot(t, 1e5*df_result[col_Name_expt_data], color=colors[idx], label=f'{int(value)} {unit}')
                if include_fits:
                    plt.plot(t, 1e5*df_result[col_Name_fit], color=colors[idx], linestyle='--', label='Fit')
                plt.ylabel(r'$10^5 \times \Delta T$ [a.u.]')

            elif MinMaxScaling:
                data_expt = MinMaxScaler(df_result[col_Name_expt_data])
                data_fit = MinMaxScaler(df_result[col_Name_fit])
                plt.plot(t, data_expt, color=colors[idx], label=f'{int(value)} {unit}')
                if include_fits:
                    plt.plot(t, data_fit, color=colors[idx], linestyle='--', label='Fit')
                plt.ylabel(r'$\Delta T$ [normalized]')
            else:
                plt.plot(t, df_result[col_Name_expt_data], color=colors[idx], label=f'{int(value)} {unit}')
                if include_fits:
                    plt.plot(t, df_result[col_Name_fit], color=colors[idx], linestyle='--', label='Fit')
                plt.ylabel(r'$\Delta T$ [a.u.]')

            plt.xlabel('t [ps]')
            plt.legend()
            plt.tight_layout()
            if save_file_name is not None:
                plt.savefig(f'{save_file_name}_{int(int(value))}_exp_Fit.pdf')
            plt.show()

        -----

        - Custom colors (red for \(A_1\) and blue for \(A_2\)) are used for distinction.
        - Absolute values of \(A_1\), \(A_2\), and their uncertainties are used to avoid issues with the logarithmic scale.

        Examples
        --------
        >>> # Example 1: Plot \(A_1\) and \(A_2\) with default settings
        >>> plot_A1_A2_semilog_together(df_param)
        >>> # Example 2: Save the plot to a specified filepath
        >>> plot_A1_A2_semilog_together(df_param, save_filepath='results/')
        >>> # Example 3: Customize the variable name and unit
        >>> plot_A1_A2_semilog_together(df_param, variable_name='Temperature', unit='K')
        """
        variable_value = df_param[variable_name]
        if scale_fluence:
            variable_value = PowerDependentResults().calculate_fluence(variable_value)
            variable_name = 'Fluence'
        A1 = df_param['A1']; A2 = df_param['A2']
        uncertainty1 = df_param['A1_uncertainty']; uncertainty2 = df_param['A2_uncertainty']

        # Define custom colors using RGB values
        red_color = (179/255, 27/255, 27/255); blue_color = (0/255, 70/255, 127/255)

        # Convert to absolute values to ensure no negative values for log scale

        uncertainty1 = np.abs(uncertainty1); uncertainty2 = np.abs(uncertainty2)

        fig, ax1 = plt.subplots()

        plt.legend(); fig.tight_layout()

        if save_filepath is not None:
            plt.savefig(f'{save_filepath}A1_A2_semilog_together.pdf')
        plt.show()

    def plot_tau1_tau2_semilog_together(df_param, unit = 'mW', variable_name = 'Power', save_filepath = None, scale_fluence = False):
        """
        Plot \(\\tau_1\) and \(\\tau_2\) on a semilogarithmic scale with asymmetric error bars for the given variable.

        Parameters
        ----------
        df_param : pandas.DataFrame
            DataFrame containing the parameters and uncertainties.

            - `variable_name` (e.g., 'Power', 'Temperature').
            - 'tau1' and 'tau2' for time constants.
            - 'tau1_uncertainty' and 'tau2_uncertainty' for the associated uncertainties.

            Default 'mW', Unit of the variable (e.g., 'mW' for power, 'K' for temperature).
        variable_name : str
            Default 'Power', Name of the variable being analyzed (e.g., 'Power', 'Temperature').
        save_filepath : str, optional
            Filepath to save the resulting plot as a PDF. If `None`, the plot is not saved.

        Notes
        ------
        - Custom colors are used for distinction: red for \(\\tau_1\) and blue for \(\\tau_2\).
        - Error bars are calculated asymmetrically to prevent negative values on the logarithmic scale.
        - The y-axis is set to a logarithmic scale for better visualization of time constants over multiple orders of magnitude.

        Examples
        --------
        >>> # Example 1: Plot \(\\tau_1\) and \(\\tau_2\) with default settings
        >>> plot_tau1_tau2_semilog_together(df_param)
        >>> # Example 2: Save the plot to a specified filepath
        >>> plot_tau1_tau2_semilog_together(df_param, save_filepath='results/')
        >>> # Example 3: Customize the variable name and unit
        >>> plot_tau1_tau2_semilog_together(df_param, variable_name='Temperature', unit='K')
        """

        if scale_fluence:
            variable_value = PowerDependentResults().calculate_fluence(variable_value)
            variable_name = 'Fluence'
        tau1 = df_param['tau1']; tau2 = df_param['tau2']
        uncertainty1 = df_param['tau1_uncertainty']; uncertainty2 = df_param['tau2_uncertainty']

        # Define custom colors using RGB values
        red_color = (179/255, 27/255, 27/255)
        blue_color = (0/255, 70/255, 127/255)

        # Calculate asymmetric error bars for log scale
        tau1_lower_error = tau1 - np.maximum(tau1 - uncertainty1, 1E-2)  # Ensure no negative values
        tau1_upper_error = uncertainty1
        tau2_lower_error = tau2 - np.maximum(tau2 - uncertainty2, 1E-2)
        tau2_upper_error = uncertainty2

        fig, ax1 = plt.subplots()
        # Plot tau1 on the left y-axis with a semilog scale
        ax1.errorbar(variable_value, tau1, yerr=[tau1_lower_error, tau1_upper_error],
                    fmt='<--', color=red_color, label=r'$\tau_1$')

                    fmt='>--', color=blue_color, label=r'$\tau_2$')
        
        ax1.set_yscale('log')
        ax1.set_xlabel(f'{variable_name} [{unit}]'); ax1.set_ylabel(r'$\tau$ [ps]')

    @staticmethod
    def PumpProbe_3_panel_plot_variable(df_result, df_param, grouping, save_filepath=None,
                                        MinMaxScaling=False, scale_amplitude=False, time_scale=True,
                                        include_fits=False, scale_fluence=False, colors=None,
                                        variable_name='Temperature', unit='K'):
        """
        Create a three-panel plot of pump-probe data grouped by a specified variable.

        Parameters
        ----------
        df_result : pandas.DataFrame
            DataFrame containing time-resolved experimental data and fits.
        df_param : pandas.DataFrame
            DataFrame containing parameters for grouping and shifting data.
        grouping : list of int
            Group sizes for the three panels.
        save_filepath : str, optional
            Path to save the output figure. Defaults to None.
        MinMaxScaling : bool, optional
            If True, scales data between 0 and 1. Defaults to False.
        scale_amplitude : bool, optional
            If True, scales data by 1e5. Defaults to False.
        time_scale : bool, optional
            If True, aligns data by subtracting `t0` values. Defaults to True.
        include_fits : bool, optional
            If True, includes fit data in the plots. Defaults to True.
        scale_fluence : bool, optional
            If True, converts variable to fluence. Defaults to False.
        colors : list, optional
            List of colors for the plots. Defaults to None.
        variable_name : str, optional
            Name of the variable to group by. Defaults to 'Temperature'.
        unit : str, optional
            Unit of the variable. Defaults to 'K'.
        """

        [num_1, num_2] = grouping
        variable_value = df_param[variable_name].values

        # Generate colors if not provided
        if colors is None:
            colors = cm.jet(np.linspace(0, 1, len(variable_value)))

        col_Name_expt_data = []
        col_Name_fit = []
        for idx, value in enumerate(variable_value):
            column_name = f"del_T ({int(value)}{(unit)})"
            if column_name not in df_result.columns:
                raise KeyError(f"Column {column_name} not found in df_result")
            col_Name_expt_data.append(column_name)
            col_Name_fit.append(f'{column_name}_fit')

        # Adjust variable if scaling fluence
        if scale_fluence:
            variable_value = PowerDependentResults().calculate_fluence(variable_value)  # Replace with actual fluence calculation
            variable_value = np.round(variable_value, 2)
            variable_name = 'Fluence'
            unit = r'$\mu J /cm^2$'

        t0_values = df_param['t0'].values

        # Divide variable_value into groups for each subplot
        group = [variable_value[:num_1], variable_value[num_1:num_1 + num_2], variable_value[num_1 + num_2:]]

        # Prepare figure
        fig, axs = plt.subplots(3, 1, sharex=True)
        data_linestyles = ['solid', 'dashed', 'dashdot']
        fit_linestyles = ['dashed', 'dashdot', 'solid']

        # Add legends
        all_handles, all_labels = [], []
        for subplot_idx, variable_group in enumerate(group):
            ax = axs[subplot_idx]

            for i, value in enumerate(variable_group):
                index = np.where(variable_value == value)[0][0]
                color = colors[index]  # Use consistent color for data and fit
                data_linestyle = data_linestyles[i % len(data_linestyles)]
                fit_linestyle = fit_linestyles[i % len(fit_linestyles)]

                t = df_result[f'time_delay [ps]({value}{unit})']
                data_exp = df_result[col_Name_expt_data[index]]
                fit = df_result[col_Name_fit[index]]
                t0 = t0_values[index] if time_scale else 0
                t = t - t0
                ax.set_xlim(min(t) + 2, max(t))
                if MinMaxScaling:
                    plot_exp, = ax.plot(t, MinMaxScaler(data_exp), label=f'{value}{unit}', color=color, linestyle=data_linestyle)
                    if include_fits:
                        plot_fit, = ax.plot(t, MinMaxScaler(fit), color=color, linestyle=fit_linestyle, label=f'{value} Fit')
                elif scale_amplitude:
                    plot_exp, = ax.plot(t, 1e5 * data_exp, label=f'{value}{unit}', color=color, linestyle=data_linestyle)
                    if include_fits:
                        plot_fit, = ax.plot(t, 1e5 * fit, color=color, linestyle=fit_linestyle, label=f'{value} Fit')
                else:
                    plot_exp, = ax.plot(t, data_exp, label=f'{value}{unit}', color=color, linestyle=data_linestyle)
                    if include_fits:
                        plot_fit, = ax.plot(t, fit, color=color, linestyle=fit_linestyle, label=f'{value} Fit')

                # Collect handles and labels
                all_handles.append(plot_exp)
                all_labels.append(f'{value} {unit}')
                if include_fits:
                    all_handles.append(plot_fit)
                    all_labels.append(f'{value} Fit')

            ax.yaxis.set_tick_params(labelcolor='none')  # Remove y values from subplots.

        # Add a single combined legend
        fig.legend(handles=all_handles, labels=all_labels,
                   bbox_to_anchor=(0.69, 0.915), loc='upper left', borderaxespad=0., labelspacing=0.4)
        # fig.legend(handles=all_handles, labels=all_labels, title=f'{variable_name} [{unit}]',
        #            bbox_to_anchor=(0.8, 0.91), loc='upper left', borderaxespad=0., labelspacing=0.4)

        # Add a single y-axis label
        if MinMaxScaling:
            fig.text(0.0, 0.5, r'$\Delta T$ (normalized)', va='center', rotation='vertical')
        else:
            fig.text(0.0, 0.5, r'$\Delta T$ [a.u.]', va='center', rotation='vertical')

        # Set x-axis label only for the bottom subplot
        axs[-1].set_xlabel(r'$t$ [ps]')

        # Adjust layout to include all elements
        plt.tight_layout(rect=[0, 0, 0.75, 1])  # Shrink the plot area to leave space for the legend

        plt.subplots_adjust(hspace=0)

        # Save the figure if specified
        if save_filepath:
            plt.savefig(f"{save_filepath}_3_panel_plot.pdf")
        plt.show()

def MinMaxScaler(data):
    '''
    Scale Data in range [-1, 0].

    Parameters
    ----------
    data : np.array
        Data to be scaled.

    Returns
    -------
    np.array
        Scaled data.
    '''
    return ( data - min(data))/ ( max(data) - min(data)) - 1
